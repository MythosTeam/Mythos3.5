update_purity_to_next = {
	set_variable = { which = purity_to_next which = purity_level }
	if = {
		limit = { check_variable = { which = purity_level value = 0 } }
		change_variable = { which = purity_to_next value = 1 }
		multiply_variable = { which = purity_to_next value = 500 } # If too hard, try 500
	}
	else = {
		multiply_variable = { which = purity_to_next value = -500 }
	}
}

update_purity_level = {
	update_purity_to_next = yes
	if = {
		limit = { is_variable_equal = { which = purity_level value = 10 } }
		if = {
			limit = { check_variable = { which = purity_progress value = 500 } }
			set_variable = { which = purity_progress value = 500 }
		}
	}
	if = {
		limit = { check_variable = { which = purity_progress which = purity_to_next } }
		change_variable = { which = purity_level value = 1 }
		subtract_variable = { which = purity_progress which = purity_to_next }
		update_purity_to_next = yes
	}
	else_if = {
		limit = { NOT = { check_variable = { which = purity_progress value = 0 } } }
		subtract_variable = { which = purity_level value = 1 }
		update_purity_to_next = yes
		change_variable = { which = purity_progress which = purity_to_next }
	}
	if = {
		limit = { NOT = { check_variable = { which = purity_level value = -10 } } }
		set_variable = { which = purity_level value = -10 }
		set_variable = { which = purity_progress value = 0 }
		update_purity_to_next = yes
	}
}

update_purity_progress_percent = {
	set_variable = { which = purity_progress_percent which = purity_progress }
	multiply_variable = { which = purity_progress_percent value = 100 }
	divide_variable = { which = purity_progress_percent which = purity_to_next }
}
