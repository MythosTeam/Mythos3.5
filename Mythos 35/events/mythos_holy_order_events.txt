	is_triggered_only = yes
	
	immediate = {
		any_title_under = {
			limit = {
				tier = BARON
				location = {
					owner = {
						OR = {
							character = FROM
							is_liege_or_above = FROM
						}
					}
				}
			}
			usurp_title = FROM
		}
		
		if = {
			limit = {
				is_liege_or_above = FROM
			}

		}

			letter_event = {
				id = MythosHolyOrders.21
			}
		}
		
		opinion = {
			who = FROM
			modifier = opinion_expelled_from_realm
		}
	}
}

# The Grand Master demands one of your sons due to a debt
character_event = {
	id = MythosHolyOrders.23
	hide_window = yes
	
	min_age = 16
	only_playable = yes
	religion_group = pagan_group
	
	trigger = {
		OR = {
			religion = draconic_pagan_reformed
			religion = draconic_pagan
		}
		is_heretic = no
		has_character_modifier = borrowed_from_golden_order
		NOT = { has_character_flag = sent_son_to_golden_order }
		any_child = {
			is_female = no
			is_adult = yes
			is_ruler = no
			prisoner = no
			NOT = { trait = incapable }
			OR = { # Prefers dragons but will accept skilled humans
				trait = dragon
				trait = dragonborn
				martial = 15
			}
			ROOT = {
				NOT = {
					current_heir = {
						character = PREVPREV
					}
				}
			}
		}
		has_dlc = "Sons of Abraham"
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	immediate = {
		random_child = {
			limit = {
				is_female = no
				is_adult = yes
				is_ruler = no
				prisoner = no
				NOT = { trait = incapable }
				OR = { # Prefers dragons but will accept skilled humans
					trait = dragon
					trait = dragonborn
					martial = 15
				}
				ROOT = {
					NOT = {
						current_heir = {
							character = PREVPREV
						}
					}
				}
			}
			character_event = {
				id = MythosHolyOrders.24
			}
		}
	}
}

		}

		piety = 100
		
		d_golden_order = {
			holder_scope = {
				FROMFROM = {
					hidden_tooltip = {
						clear_wealth = PREV
					}
					spouse = {
						remove_spouse = spouse
						hidden_tooltip = { character_event = { id = SoA.2862 } }
					}
					move_character = PREV
					add_trait = celibate
					end_inaccessibility_effect = yes #removes in_hiding or in_seclusion

				}
			}
		}
	}
	
	option = {
		name = EVTOPTBMythosHolyOrders.26 # Pay them back
		wealth = -300
		hidden_tooltip = {
			remove_character_modifier = borrowed_from_golden_order
		}
		FROM = {
			wealth = 300
		}
	}
	
	option = {
		name = EVTOPTCMythosHolyOrders.26 # Refuse
		piety = -1000
	}

# The Grand Master demands the right to build a castle in your demesne

	id = MythosHolyOrders.28

	only_playable = yes
	religion_group = pagan_group

		}

		has_character_modifier = borrowed_from_golden_order
		NOT = { has_character_flag = sent_son_to_golden_order }
		
		any_demesne_province = {
			has_empty_holding = yes
		}
		
		d_golden_order = {
			holder_scope = {
				wealth = 300
			}
		}
		has_dlc = "Sons of Abraham"
		
		NOT = { has_character_modifier = expelled_d_golden_order }
		NOT = {
			any_liege = {
				has_character_modifier = expelled_d_golden_order
			}
		}
	}
