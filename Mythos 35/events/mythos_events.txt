# Gain Philosopher's Stone through inheritance
character_event = {
	id = MYTH.54
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		FROM = {
			artifact_type = philosopher_stone
		}
	}
	
	immediate = {
		add_trait = philosopher_stone_trait
	}
}

# Clear Philosopher's Stone trait
character_event = {
	id = MYTH.55
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { 
				has_artifact = philosopher_stone
				NOT = { trait = philosopher_stone_trait }
			}
			add_trait = philosopher_stone_trait
		}
		else_if = {
			limit = { 
				NOT = { has_artifact = philosopher_stone }
				trait = philosopher_stone_trait
			}
			remove_trait = philosopher_stone_trait
		}
	}
}

# Recalculate Virtues count
character_event = {
	id = MYTH.56
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		count_personal_virtues = yes
		count_personal_vices = yes
		set_variable = { which = virtue_vice_difference which = num_virtues }
		subtract_variable = { which = virtue_vice_difference which = num_vices }
		
		if = {
			limit = { is_variable_equal = { which = virtue_vice_difference value = 0 } }
			clear_vice_virtue_modifiers = yes
		}
		else_if = {
			limit = {
				is_variable_equal = { which = virtue_vice_difference value = 1 }
			}
			clear_vice_virtue_modifiers = yes
			add_character_modifier = { name = virtuous_1 duration = -1 }
		}
		else_if = {
			limit = {
				is_variable_equal = { which = virtue_vice_difference value = 2 }
